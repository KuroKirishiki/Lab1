
Sub z1()
Dim a(15) As Byte, i As Byte, k As Byte, x As Byte, n As Byte, s As Integer, sra As Double, s1 As String, s2 As String, fl As Boolean
k = 0
n = 0
sra = 0
s = 0
For i = 0 To 14
    Do While a(i) = 0 /заполнение массива чтобы убрать нули
     x = Int(Rnd * 100)
     If x < 25 Then
      a(i) = x
     End If
    Loop
    If a(i) = 2 Then смотрим наличие хотя бы одной двойки
     fl = True
    End If
    s1 = s1 & a(i) & "  " записываем первый массив
Next i

MsgBox (fl)

If fl = True Then есть ли двойка
    For i = 0 To 14
        If a(i) Mod 2 = 0 Then если чётное то обнуляем и прибавляем в счётчике +1
         a(i) = 0
         k = k + 1
        End If
        s2 = s2 & a(i) & "  " записываем во второй массив
    Next i
    MsgBox (s1 & vbCrLf & s2 & vbCrLf & k)
Else:
    For i = 0 To 14
        If a(i) Mod 2 <> 0 Then если число нечётное то считаем сумму и счётчик +1
         s = s + a(i)
         n = n + 1
        End If
        s2 = s2 & a(i) & "  " записываем второй массив
    Next i
    If n > 0 Then
        sra = s / n считаем среднее арифмитичемское если нечётные были
        MsgBox (s1 & vbCrLf & s2 & vbCrLf & sra)
    End If
End If
End Sub
